Range with config

someRange2config: any = {
  behaviour: 'drag',
  connect: true,
  margin: 1,
  limit: 5, // NOTE: overwritten by [limit]="10"
  range: {
    min: 0,
    max: 20
  },
  pips: {
    mode: 'steps',
    density: 5
  }
};
<nouislider [config]="someRange2config" [limit]="10" [(ngModel)]="someRange2"></nouislider>

Keyboard support

someKeyboardConfig: any = {
  behaviour: 'drag',
  connect: true,
  start: [0, 5],
  step: 0.1,
  keyboard: true,  // same as [keyboard]="true"
  range: {
    min: 0,
    max: 5
  },
  pips: {
    mode: 'count',
    density: 2,
    values: 6,
    stepped: true
  }
};

<nouislider [config]="someKeyboardConfig" [(ngModel)]="someKeyboard"></nouislider>
Select a handle and use arrow keys or page up / page down.
